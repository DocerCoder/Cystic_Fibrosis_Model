  <script>
    // Simple, transparent, educational scoring (not medical).
    // Risk: based on genotype cues and sweat chloride, modulated by FEV1 and infections.
    // Severity: weighted mix of FEV1, infections, BMI, comorbidities.
    const fields = [
      "variant1","variant2","sweat","fev1","age","sex","bmi",
      "modulator","pseudomonas","ntm","cfrd","pi"
    ];

    function val(id){ return (document.getElementById(id)?.value || "").trim(); }

    function pctCompleteness(){
      const filled = fields.reduce((a,f)=> a + (val(f)?1:0), 0);
      return Math.round(100 * filled / fields.length);
    }

    function eligFromGenotype(v1, v2){
      const g = (v1 + " " + v2).toLowerCase();
      if(!g.trim()) return "Insufficient genotype data to assess potential modulator eligibility.";
      if(g.includes("g551d")) return "Likely responsive to ivacaftor-class modulators (Gating mutation).";
      if(g.includes("f508del") && (g.match(/f508del/g)||[]).length >= 2) return "F508del homozygous: often eligible for ETI (elexacaftor/tezacaftor/ivacaftor).";
      if(g.includes("f508del")) return "F508del heterozygous: ETI may be considered depending on second allele.";
      if(/w1282x|g542x|r553x|n1303k/.test(g)) return "Nonsense/Class I variants: response depends on specific modulator indications.";
      return "Genotype present but not matched to common modulator rules in this demo.";
    }

    function compute(){
      const v1 = val("variant1"), v2 = val("variant2");
      const sweat = Number(val("sweat") || NaN);
      const fev1 = Number(val("fev1") || NaN);
      const bmi = Number(val("bmi") || NaN);
      const pseudo = val("pseudomonas");
      const ntm = val("ntm");
      const cfrd = val("cfrd")==="Yes";
      const pi = val("pi")==="Yes";
      const mod = val("modulator");

      // Completeness & confidence
      const comp = pctCompleteness();
      const confidence = Math.round(50 + 0.5 * comp); // 50–100
      document.getElementById("completeness").textContent = `Completeness: ${comp}%`;
      document.getElementById("confidence").textContent = `${confidence}%`;

      // Risk score (0–100)
      let risk = 20;
      const g = (v1+" "+v2).toLowerCase();
      if(g.includes("f508del")) risk += 20;
      if((g.match(/f508del/g)||[]).length>=2) risk += 10;
      if(/g551d|w1282x|g542x|r553x|n1303k/.test(g)) risk += 10;
      if(!isNaN(sweat)){
        if(sweat >= 60) risk += 25;
        else if(sweat >= 30) risk += 10;
        else risk -= 10;
      }
      if(!isNaN(fev1)){
        if(fev1 < 60) risk += 10;
        else if(fev1 >= 90) risk -= 5;
      }
      if(pseudo === "Chronic") risk += 5;
      risk = Math.max(0, Math.min(100, Math.round(risk)));

      // Severity index (0–100): lower is better
      let sev = 50;
      if(!isNaN(fev1)){
        sev += (80 - fev1) * 0.5; // each % below 80 increases severity
      }
      if(pseudo === "Chronic") sev += 10;
      if(pseudo === "Intermittent") sev += 5;
      if(ntm && ntm !== "") sev += 5;
      if(!isNaN(bmi)){
        if(bmi < 18.5) sev += 8;
        else if(bmi >= 23) sev -= 5;
      }
      if(cfrd) sev += 6;
      if(pi) sev += 4;
      if(mod && mod !== "") sev -= 6; // on-modulator improvement signal
      sev = Math.max(0, Math.min(100, Math.round(sev)));

      // Populate UI
      document.getElementById("riskScore").textContent = risk;
      document.getElementById("severityIndex").textContent = sev;
      document.getElementById("sevBar").style.width = sev + "%";

      // Eligibility text
      document.getElementById("eligibility").textContent = eligFromGenotype(v1, v2);

      // Feature contributions (illustrative)
      const explain = [
        {name:"Genotype pattern", val: (g.includes("f508del")?"+":"") + (g.includes("g551d")?"+":"") || "±", score: (g.includes("f508del")?+12:0)+(g.includes("g551d")?+8:0)},
        {name:"Sweat chloride", val: isNaN(sweat)?"—":sweat+" mmol/L", score: isNaN(sweat)?0:(sweat>=60?+18:(sweat>=30?+8:-6))},
        {name:"FEV₁ % predicted", val: isNaN(fev1)?"—":fev1+"%", score: isNaN(fev1)?0:(fev1<60?+12:(fev1>=90?-6:-2))},
        {name:"Pseudomonas status", val: pseudo||"—", score: pseudo==="Chronic"?+8:(pseudo==="Intermittent"?+4:0)},
        {name:"Nutritional status (BMI)", val: isNaN(bmi)?"—":bmi, score: isNaN(bmi)?0:(bmi<18.5?+6:(bmi>=23?-4:0))},
        {name:"Comorbidities (CFRD/PI)", val: (cfrd?"CFRD ":"")+(pi?"PI":"") || "—", score: (cfrd?+4:0)+(pi?+3:0)},
        {name:"On CFTR modulator", val: mod||"No", score: mod? -6 : 0},
      ];

      const list = document.getElementById("explainList");
      list.innerHTML = "";
      explain.forEach(e=>{
        const div = document.createElement("div");
        div.className = "item";
        const color = e.score>5?"#fca5a5":(e.score<-3?"#86efac":"#cbd5e1");
        div.innerHTML = `<div class="name">${e.name}<div class="hint">${e.val}</div></div>
                         <div class="score" style="color:${color}">${e.score>0? "+"+e.score : e.score}</div>`;
        list.appendChild(div);
      });
    }

    document.getElementById("computeBtn").addEventListener("click", compute);
    document.getElementById("resetBtn").addEventListener("click", () => {
      [...document.querySelectorAll("input,select,textarea")].forEach(el=> el.value = "");
      document.getElementById("riskScore").textContent = "—";
      document.getElementById("severityIndex").textContent = "—";
      document.getElementById("sevBar").style.width = "0%";
      document.getElementById("confidence").textContent = "—";
      document.getElementById("completeness").textContent = "Completeness: 0%";
      document.getElementById("eligibility").textContent = "—";
      document.getElementById("explainList").innerHTML = "";
    });
  </script>
