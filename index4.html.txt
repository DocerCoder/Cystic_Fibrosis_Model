    const KNOWN_CFTR = new Set([
      "f508del","g551d","w1282x","g542x","r553x","n1303k","r117h",
      "3849+10kb c→t","3849+10kb c>t","d1152h","2789+5g→a","2789+5g>a","a455e",
      "g85e","1717-1g→a","1717-1g>a","621+1g→t","621+1g>t","2t>g","p67l","g178r"
    ]);
    function detectRarity(g1, g2){
      const u1 = (g1||"").trim().toLowerCase();
      const u2 = (g2||"").trim().toLowerCase();
      const v1Known = u1 && KNOWN_CFTR.has(u1);
      const v2Known = u2 && KNOWN_CFTR.has(u2);
      if (!u1 && !u2) return {autoRare:false, reason:"no genotype provided"};
      const autoRare = !(v1Known || v2Known);
      return {autoRare, reason: autoRare ? "not in known catalog" : "known catalog match"};
    }

    // --- Scoring (weight-aware) ---
    function computeScores(inputs){
      const {sweat, fev1, bmi, pseudo, ntm, cfrd, pi, mod, g1, g2} = inputs;

      let risk = 20;
      if (sweat !== null){ risk += WEIGHTS.sweat * (sweat>=60?25:(sweat>=30?10:-10)); }
      if (fev1 !== null){ risk += WEIGHTS.fev1  * (fev1<60?10:(fev1>=90?-5:0)); }
      if (pseudo === "Chronic") risk += WEIGHTS.pseudo * 6;
      if (/f508del/.test(g1+g2)) risk += 10;
      if (/g551d|w1282x|g542x|r553x|n1303k/.test((g1+g2))) risk += 6;
      risk = clamp(Math.round(risk),0,100);

      let sev = 50;
      if (fev1 !== null) sev += WEIGHTS.fev1 * (80 - fev1) * 0.5;
      if (pseudo === "Chronic") sev += WEIGHTS.pseudo * 10;
      else if (pseudo === "Intermittent") sev += WEIGHTS.pseudo * 5;
      if (ntm && ntm !== "") sev += 5;
      if (bmi !== null){ if (bmi < 18.5) sev += WEIGHTS.bmi*8; else if (bmi >= 23) sev -= WEIGHTS.bmi*5; }
      if (cfrd) sev += WEIGHTS.cfrd*6;
      if (pi) sev += WEIGHTS.pi*4;
      if (mod) sev -= WEIGHTS.mod*6;
      sev = clamp(Math.round(sev),0,100);

      return {risk, sev};
    }

    // --- Genotype class inference from phenotype (weight-aware) ---
    function inferGenotypeClass(inputs){
      const {sweat, fev1, pseudo, bmi, cfrd, pi} = inputs;
      let score = 0;
      const signals = [];

      if (sweat !== null){
        if (sweat >= 90){ score += 3*WEIGHTS.sweat; signals.push("very high sweat chloride"); }
        else if (sweat >= 60){ score += 2*WEIGHTS.sweat; signals.push("high sweat chloride"); }
        else if (sweat >= 30){ score += 1*WEIGHTS.sweat; signals.push("intermediate sweat chloride"); }
        else { score -= 2*WEIGHTS.sweat; signals.push("low sweat chloride"); }
      }
      if (fev1 !== null){
        if (fev1 < 40){ score += 3*WEIGHTS.fev1; signals.push("very low FEV₁"); }
        else if (fev1 < 60){ score += 2*WEIGHTS.fev1; signals.push("low FEV₁"); }
        else if (fev1 < 80){ score += 1*WEIGHTS.fev1; signals.push("moderate FEV₁"); }
        else if (fev1 >= 90){ score -= 1*WEIGHTS.fev1; signals.push("high FEV₁"); }
      }
      if (pseudo){
        if (pseudo === "Chronic"){ score += 2*WEIGHTS.pseudo; signals.push("chronic Pseudomonas"); }
        else if (pseudo === "Intermittent"){ score += 1*WEIGHTS.pseudo; signals.push("intermittent Pseudomonas"); }
        else if (pseudo === "None"){ score -= 1*WEIGHTS.pseudo; signals.push("no Pseudomonas"); }
      }
      if (bmi !== null){
        if (bmi < 18.5){ score += 1*WEIGHTS.bmi; signals.push("low BMI"); }
        else if (bmi >= 23){ score -= 1*WEIGHTS.bmi; signals.push("higher BMI"); }
      }
      if (cfrd){ score += 1*WEIGHTS.cfrd; signals.push("CFRD"); }
      if (pi){ score += 1*WEIGHTS.pi; signals.push("pancreatic insufficiency"); }

      let klass, examples, msev;
      if (score >= 8){ klass = "Class I–II likely"; examples = "e.g., F508del/F508del, W1282X, G542X"; msev = 80; }
      else if (score >= 5){ klass = "Class II–III possible"; examples = "e.g., F508del/other, G551D"; msev = 55; }
      else { klass = "Class IV–V or atypical"; examples = "e.g., R117H, 3849+10kb C→T"; msev = 25; }

      const used = (sweat!==null) + (fev1!==null) + (!!pseudo) + (bmi!==null) + (!!cfrd) + (!!pi);
      const strength = Math.min(6, Math.abs(score - 6));
      const confidence = clamp(Math.round(48 + used*8 + strength*3), 50, 96);

      return {klass, examples, confidence, signals, mutationSeverity:msev};
    }

    // --- If genotype provided, refine class ---
    function refineWithGenotype(g1,g2,inferred){
      const g = (g1+" "+g2).toLowerCase();
      if (!g.trim()) return inferred;
      if (g.includes("f508del") && (g.match(/f508del/g)||[]).length>=2)
        return {klass:"Class II (processing) — strong", examples:"F508del/F508del", confidence:Math.max(inferred.confidence, 90), signals:[...inferred.signals, "provided genotype"], mutationSeverity:85};
      if (g.includes("g551d"))
        return {klass:"Class III (gating) — strong", examples:"G551D/other", confidence:Math.max(inferred.confidence, 88), signals:[...inferred.signals, "provided genotype"], mutationSeverity:60};
      if (/w1282x|g542x|r553x|n1303k/.test(g))
        return {klass:"Class I (nonsense) — strong", examples:"e.g., W1282X/other", confidence:Math.max(inferred.confidence, 88), signals:[...inferred.signals, "provided genotype"], mutationSeverity:78};
      if (g.includes("r117h"))
        return {klass:"Class IV (conductance) — moderate", examples:"R117H/other", confidence:Math.max(inferred.confidence, 80), signals:[...inferred.signals, "provided genotype"], mutationSeverity:35};
      return {...inferred, signals:[...inferred.signals, "genotype noted (unmapped demo)"], confidence:Math.max(inferred.confidence, 75)};
    }

    // --- Eligibility narrative ---
    function modulatorEligibility(g1,g2,inferred){
      const g = (g1+" "+g2).toLowerCase();
      if (g.includes("g551d")) return "Ivacaftor-responsive pattern (gating) likely.";
      if (g.includes("f508del") && (g.match(/f508del/g)||[]).length>=2) return "ETI likely appropriate (F508del homozygous).";
      if (g.includes("f508del")) return "Possible ETI eligibility (F508del heterozygous; depends on second allele).";
      if (/w1282x|g542x|r553x|n1303k/.test(g)) return "Class I variants: modulator response varies by indication.";
      if (inferred.klass.startsWith("Class I") || inferred.klass.startsWith("Class II")) return "Phenotype suggests loss-of-function; ETI class benefits plausible if genotype aligns.";
      if (inferred.klass.startsWith("Class III")) return "Gating features; ivacaftor-class benefit plausible if genotype confirms.";
      if (inferred.klass.startsWith("Class IV") || inferred.klass.startsWith("Class V")) return "Milder classes; benefit may be genotype-specific.";
      return "Insufficient information to estimate eligibility.";
    }

    // --- Feature breakdown for UI ---
    function featuresBreakdown(inputs){
      const {sweat, fev1, bmi, pseudo} = inputs;
      const cfrdYes = (val('cfrd')==="Yes");
      const piYes = (val('pi')==="Yes");
      const modSel = val('modulator');

      const sweatScore = (sweat===null?0:(sweat>=60?+18:(sweat>=30?+8:-6))) * WEIGHTS.sweat;
      const fev1Score  = (fev1===null?0:(fev1<60?+12:(fev1>=90?-6:-2))) * WEIGHTS.fev1;
      const pseudoScore= (pseudo==="Chronic"?+8:(pseudo==="Intermittent"?+4:0)) * WEIGHTS.pseudo;
      const bmiScore   = (bmi===null?0:(bmi<18.5?+6:(bmi>=23?-4:0))) * WEIGHTS.bmi;
      const comorbScore= (cfrdYes?+4:0)*WEIGHTS.cfrd + (piYes?+3:0)*WEIGHTS.pi;
      const modScore   = (modSel? -6 : 0) * WEIGHTS.mod;

      const sh = num('shannon_diversity');
      const phy = val('dominant_phylum');
      const mbFeat = microbiomePenalty({ shannon_diversity: sh, dominant_phylum: phy || null });

      const list = [
        {name:"Sweat chloride", val: (sweat===null?"—":sweat+" mmol/L"), score: Math.round(sweatScore)},
        {name:"FEV₁ % predicted", val: (fev1===null?"—":fev1+"%"), score: Math.round(fev1Score)},
        {name:"Pseudomonas status", val: val('pseudomonas')||"—", score: Math.round(pseudoScore)},
        {name:"Nutritional status (BMI)", val: (bmi===null?"—":bmi), score: Math.round(bmiScore)},
        {name:"Comorbidities (CFRD/PI)", val: ((cfrdYes?"CFRD ":"") + (piYes?"PI":"")) || "—", score: Math.round(comorbScore)},
        {name:"On CFTR modulator", val: val('modulator')||"No", score: Math.round(modScore)},
        {name:"Microbiome", val: (sh==null?"—":`Shannon ${sh}`) + (phy?` • ${phy}`:""), score: Math.round(mbFeat)}
      ];
      return list;
    }
    function renderExplain(list){
      const el = get("explainList");
      el.innerHTML = "";
      list.forEach(e=>{
        const div = document.createElement("div");
        div.className = "item";
        const color = e.score>5 ? "#ef4444" : (e.score<-3 ? "#34d399" : "#1f2937");
        div.innerHTML = `<div class="name">${e.name}<div class="help">${e.val}</div></div>
                         <div class="score" style="color:${color}">${e.score>0? "+"+e.score : e.score}</div>`;
        el.appendChild(div);
      });
    }

      const tbody = get("summaryTable").querySelector("tbody");

      const mb = readMicrobiomeInputs();

      const rows = [
        ["Sweat chloride", inputs.sweat ?? "—"],
        ["FEV₁ % predicted", inputs.fev1 ?? "—"],
        ["Pseudomonas", inputs.pseudo || "—"],
        ["BMI", inputs.bmi ?? "—"],
        ["CFRD", inputs.cfrd ? "Yes" : (val('cfrd')===""?"Unknown":"No")],
        ["Pancreatic insufficiency", inputs.pi ? "Yes" : (val('pi')===""?"Unknown":"No")],
        ["Modulator", inputs.mod || "None"],
        ["Microbiome diversity (Shannon)", mb.shannon_diversity ?? "—"],
        ["Dominant phylum", mb.dominant_phylum ?? "—"],

        ["Provided genotype", ((inputs.g1+inputs.g2).trim()? (inputs.g1+" / "+inputs.g2) : "None")],

        ["Examples", geno.examples],

      rows.forEach(([k,v])=>{

        tr.innerHTML = `<th class="muted" style="font-weight:600">${k}</th><td>${v}</td>`;
        tbody.appendChild(tr);
      });
    }

      if (sev>=70) return "Severe phenotype";
      if (sev>=40) return "Moderate phenotype";

    // --- A/B snapshots ---

      return {

      const state = currentState();
      SNAPSHOTS[which] = state;
      get(which==='A' ? 'snapA' : 'snapB').textContent =
        `Risk: ${state.outputs.risk} | Severity: ${state.outputs.severity}
Class: ${state.outputs.class}
Elig: ${state.outputs.eligibility}`;

    function clearSnapshots(){
      SNAPSHOTS.A = SNAPSHOTS.B = null;
      get("snapA").textContent = "—";
      get("snapB").textContent = "—";
      get("snapDelta").innerHTML = "";
    }
    function renderDelta(){
      const A = SNAPSHOTS.A, B = SNAPSHOTS.B;
      const tb = get("snapDelta"); if(!tb) return;
      tb.innerHTML = "";
      if(!A || !B) return;
      const del = (a,b)=> (a==null||b==null) ? "—" : (b - a>0? "+"+(b-a):(b-a));
      const rows = [
        ["Risk", A.outputs.risk, B.outputs.risk, del(A.outputs.risk, B.outputs.risk)],
        ["Severity", A.outputs.severity, B.outputs.severity, del(A.outputs.severity, B.outputs.severity)],
        ["Class", A.outputs.class, B.outputs.class, A.outputs.class===B.outputs.class ? "same" : "changed"],
        ["Eligibility", A.outputs.eligibility, B.outputs.eligibility, A.outputs.eligibility===B.outputs.eligibility ? "same" : "changed"]
      ];
      rows.forEach(([k,a,b,d])=>{
        const tr = document.createElement("tr");
        tr.innerHTML = `<th class="muted" style="font-weight:600">${k}</th><td>${a}</td><td>${b}</td><td>${d}</td>`;
        tb.appendChild(tr);
      });
    }

    // --- Compute pipeline ---
    function compute(){
      const g1 = val("variant1").toLowerCase();
      const g2 = val("variant2").toLowerCase();
      const inputs = {
        sweat: num("sweat"),
        fev1: num("fev1"),
        age: num("age"),
        sex: val("sex"),
        bmi: num("bmi"),
        pseudo: val("pseudomonas"),
        ntm: val("ntm"),
        cfrd: val("cfrd")==="Yes",
        pi: val("pi")==="Yes",
        mod: val("modulator"),
        g1, g2
      };

      completenessBadge();

      // Rarity detection
      const manualRare = val("rareFlag");
      const evLevel = val("evidence");
      const rarity = detectRarity(g1, g2);
      const isRare = manualRare === "Yes" ? true : (manualRare === "No" ? false : rarity.autoRare);

      // Core inference
      let inferred = inferGenotypeClass(inputs);
      inferred = refineWithGenotype(g1,g2,inferred);
      const scores = computeScores(inputs);

      // Microbiome adjust
      const mbInputs = readMicrobiomeInputs();
      const mbScore = microbiomePenalty(mbInputs);
      scores.sev = clamp(scores.sev + mbScore, 0, 100);

      // Confidence adjustors for rarity/evidence
      let confAdj = 0;
      if (isRare) confAdj -= 8;
      if (evLevel === "Limited") confAdj -= 5;
      else if (evLevel === "Moderate") confAdj -= 2;
      else if (evLevel === "Strong") confAdj += 2;

      // Eligibility with caveat
      let eligibilityTxt = modulatorEligibility(g1,g2,inferred);
      if (isRare) {
        eligibilityTxt += " Caution: rare/uncertain variant — eligibility depends on functional data and labeling.";
      }

      // KPIs
      get("riskScore").textContent = scores.risk;
      get("severityIndex").textContent = scores.sev;
      get("sevBar").style.width = scores.sev + "%";
      get("genoClass").textContent = inferred.klass;
      get("genoExamples").textContent = inferred.examples;
      const cVal = clamp(inferred.confidence + confAdj, 35, 98);
      get("confidence").textContent = cVal + "%";

      // Eligibility
      get("eligibility").textContent = eligibilityTxt;

      // Rarity banner
      const rb = get("rarityBanner");
      if (rb) rb.style.display = isRare ? "block" : "none";

      // Feature breakdown
      const explain = featuresBreakdown(inputs);
      renderExplain(explain);

      // Microbiome score
      get("microbiome_score").textContent = mbScore;
      const mbBadges = [];
      if (mbInputs.shannon_diversity < 2.0) mbBadges.push("Low diversity ↑ severity");
      if ((mbInputs.dominant_phylum||"").toLowerCase() === "proteobacteria") mbBadges.push("Proteobacteria dominance ↑ inflammation");
      get("microbiome_badges").innerHTML = mbBadges.map(b => `<span class="pill">${b}</span>`).join('');

      // Summary tab
      fillSummary(inputs, scores, inferred);
      get("predictedProfile").textContent = profileText(scores.sev);
      get("mutationScore").textContent = inferred.mutationSeverity;
      get("mutBar").style.width = inferred.mutationSeverity + "%";
      get("rationale").textContent = inferred.signals.join(", ");

      // Save to history
      saveHistory({
        ts: Date.now(),
        inputs: {...inputs, rareFlag: val("rareFlag"), evidence: val("evidence")},
        scores, geno: inferred, eligibility: eligibilityTxt
      });
    }
