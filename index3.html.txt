    function hashStringFast(s){
      let h=2166136261; for(let i=0;i<s.length;i++){ h^=s.charCodeAt(i); h+= (h<<1)+(h<<4)+(h<<7)+(h<<8)+(h<<24); h>>>0; }

      if (window.crypto?.subtle) {

        return Array.from(new Uint8Array(digest)).map(b=>b.toString(16).padStart(2,"0")).join("").slice(0,16);
      }
      return hashStringFast(s);
    }

    function detectRarity(g1, g2){
      const u1 = (g1||"").trim().toLowerCase();
      const u2 = (g2||"").trim().toLowerCase();
      const v1Known = u1 && KNOWN_CFTR.has(u1);
      const v2Known = u2 && KNOWN_CFTR.has(u2);
      if (!u1 && !u2) return {autoRare:false, reason:"no genotype provided"};
      const autoRare = !(v1Known || v2Known);
      return {autoRare, reason: autoRare ? "not in known catalog" : "known catalog match"};
    }

    // --- Infection penalty ---
    function infectionDynamicsPenalty(){
      const burden = parseFloat(val('infection_burden')) || 0;
      const exac = parseFloat(val('exacerbations')) || 0;
      const erad = val('eradication');
      let penalty = 0;
      penalty += (burden/100) * 8;
      penalty += Math.min(exac, 6) * 1.5;
      if (erad === 'Yes') penalty -= 2;
      return Math.round(penalty);
    }

    // --- Microbiome signals ---
    function microbiomeSignals(){
      const sh = num('shannon');
      const dom = val('dominant_taxa');
      const rich = parseInt(val('path_richness')||"0");
      const dys = parseInt(val('dysbiosis')||"0");
      const recentAbx = val('recent_abx');
      const poly = val('polymicrobial');

      let riskΔ = 0, sevΔ = 0;
      if (sh!=null){
        const low = clamp(2.0 - sh, -1.5, 2.0);
        riskΔ += W.microdiv * (low * 2.5);
        sevΔ  += W.microdiv * (low * 3.0);
      }
      if (Number.isFinite(dys)){
        const d = dys/100;
        riskΔ += W.dysbio * (d * 4.0);
        sevΔ  += W.dysbio * (d * 6.0);
      }
      if (Number.isFinite(rich)){
        const rr = clamp(rich,0,12);
        riskΔ += W.rich * (rr * 0.6);
        sevΔ  += W.rich * (rr * 0.8);
      }
      if (dom==="Pseudomonas-dominant"){ riskΔ += 4; sevΔ += 8; }
      else if (dom==="Achromobacter"){ riskΔ += 3; sevΔ += 5; }
      else if (dom==="Staph-dominant"){ riskΔ += 1.5; sevΔ += 2.5; }
      else if (dom==="Anaerobe-leaning"){ riskΔ += 1; sevΔ += 2; }
      else if (dom==="Balanced/commensal"){ riskΔ += -1; sevΔ += -2; }

      if (recentAbx==="Yes (≤30 days)"){ riskΔ -= 1; sevΔ -= 2; }
      if (poly==="Yes"){ riskΔ += 1.5; sevΔ += 2.5; }

      return { riskΔ: Math.round(riskΔ), sevΔ: Math.round(sevΔ) };
    }

    // --- Scoring / inference using phenotype + genotype + microbiome ---
    function computeScores(inputs, genoRef){
      const {sweat, fev1, bmi, pseudo, ntm, cfrd, pi, mod, g1, g2} = inputs;
      let risk = 20;
      if (sweat !== null){ risk += W.sweat * (sweat>=60?25:(sweat>=30?10:-10)); }
      if (fev1 !== null){ risk += W.fev1 * (fev1<60?10:(fev1>=90?-5:0)); }
      if (pseudo === "Chronic") risk += W.pseudo * 6;
      if (/f508del/.test(g1+g2)) risk += 10;
      if (/g551d|w1282x|g542x|r553x|n1303k/.test((g1+g2))) risk += 6;

      let sev = 50;
      if (fev1 !== null) sev += W.fev1 * (80 - fev1) * 0.5;
      if (pseudo === "Chronic") sev += W.pseudo * 10;
      else if (pseudo === "Intermittent") sev += W.pseudo * 5;
      if (ntm && ntm !== "") sev += 5;
      if (bmi !== null){ if (bmi < 18.5) sev += W.bmi * 8; else if (bmi >= 23) sev -= W.bmi * 5; }
      if (cfrd) sev += W.cfrd * 6;
      if (pi) sev += W.pi * 4;
      if (mod) sev -= 6;

      const micro = microbiomeSignals();
      risk += micro.riskΔ;
      sev  += micro.sevΔ;

      if (genoRef?.sevAnchor!=null){
        const anchor = genoRef.sevAnchor;
        sev = Math.round(0.7*sev + 0.3*anchor);
        risk = Math.round(risk + (anchor-50)/40);
      }

      sev += infectionDynamicsPenalty();
      risk = clamp(Math.round(risk),0,100);
      sev = clamp(Math.round(sev),0,100);
      return {risk, sev};
    }

    function inferGenotypeClass(inputs){
      const {sweat, fev1, pseudo, bmi, cfrd, pi} = inputs;
      let score = 0;
      const signals = [];
      if (sweat !== null){
        if (sweat >= 90){ score += 3; signals.push("very high sweat chloride"); }
        else if (sweat >= 60){ score += 2; signals.push("high sweat chloride"); }
        else if (sweat >= 30){ score += 1; signals.push("intermediate sweat chloride"); }
        else { score -= 2; signals.push("low sweat chloride"); }
      }
      if (fev1 !== null){
        if (fev1 < 40){ score += 3; signals.push("very low FEV₁"); }
        else if (fev1 < 60){ score += 2; signals.push("low FEV₁"); }
        else if (fev1 < 80){ score += 1; signals.push("moderate FEV₁"); }
        else if (fev1 >= 90){ score -= 1; signals.push("high FEV₁"); }
      }
      if (pseudo){
        if (pseudo === "Chronic"){ score += 2; signals.push("chronic Pseudomonas"); }
        else if (pseudo === "Intermittent"){ score += 1; signals.push("intermittent Pseudomonas"); }
        else if (pseudo === "None"){ score -= 1; signals.push("no Pseudomonas"); }
      }
      if (bmi !== null){
        if (bmi < 18.5){ score += 1; signals.push("low BMI"); }
        else if (bmi >= 23){ score -= 1; signals.push("higher BMI"); }
      }
      if (cfrd){ score += 1; signals.push("CFRD"); }
      if (pi){ score += 1; signals.push("pancreatic insufficiency"); }

      let klass, examples, msev;
      if (score >= 8){ klass = "Class I–II likely"; examples = "e.g., F508del/F508del, W1282X, G542X"; msev = 80; }
      else if (score >= 5){ klass = "Class II–III possible"; examples = "e.g., F508del/other, G551D"; msev = 55; }
      else { klass = "Class IV–V or atypical"; examples = "e.g., R117H, 3849+10kb C→T"; msev = 25; }

      const used = (sweat!==null) + (fev1!==null) + (!!pseudo) + (bmi!==null) + (!!cfrd) + (!!pi);
      const strength = Math.min(6, Math.abs(score - 6));
      const confidence = clamp(Math.round(48 + used*8 + strength*3), 50, 96);

      return {klass, examples, confidence, signals, mutationSeverity:msev};
    }

    function cftr2GenoReference(g1, g2){
      const toks = [g1,g2].map(s=> (s||"").toLowerCase().trim()).filter(Boolean);
      if (!toks.length) return null;
      const priors = toks.map(t=> CFTR2_PRIORS[t]).filter(Boolean);
      if (!priors.length) return null;
      const worst = priors.reduce((a,b)=> (b.sevPrior>a.sevPrior?b:a), priors[0]);
      const sevAnchor = Math.round(100*worst.sevPrior);
      return { sevAnchor, classHint: worst.class, resp: worst.resp };
    }

    function refineWithGenotype(g1,g2,inferred){
      const g = (g1+" "+g2).toLowerCase();
      if (!g.trim()) return inferred;
      if (g.includes("f508del") && (g.match(/f508del/g)||[]).length>=2)
        return {klass:"Class II (processing) — strong", examples:"F508del/F508del", confidence:Math.max(inferred.confidence, 92), signals:[...inferred.signals, "provided genotype"], mutationSeverity:88};
      if (g.includes("g551d"))
        return {klass:"Class III (gating) — strong", examples:"G551D/other", confidence:Math.max(inferred.confidence, 90), signals:[...inferred.signals, "provided genotype"], mutationSeverity:62};
      if (/w1282x|g542x|r553x|n1303k/.test(g))
        return {klass:"Class I (nonsense) — strong", examples:"e.g., W1282X/other", confidence:Math.max(inferred.confidence, 90), signals:[...inferred.signals, "provided genotype"], mutationSeverity:80};
      if (g.includes("r117h"))
        return {klass:"Class IV (conductance) — moderate", examples:"R117H/other", confidence:Math.max(inferred.confidence, 82), signals:[...inferred.signals, "provided genotype"], mutationSeverity:40};
      return {...inferred, signals:[...inferred.signals, "genotype noted (unmapped)"], confidence:Math.max(inferred.confidence, 76)};
    }

    function modulatorEligibility(g1,g2,inferred){
      const g = (g1+" "+g2).toLowerCase();
      if (g.includes("g551d")) return "Ivacaftor-responsive pattern (gating) likely.";
      if (g.includes("f508del") && (g.match(/f508del/g)||[]).length>=2) return "ETI likely appropriate (F508del homozygous).";
      if (g.includes("f508del")) return "Possible ETI eligibility (F508del heterozygous; depends on second allele).";
      if (/w1282x|g542x|r553x|n1303k/.test(g)) return "Class I variants: modulator response varies by indication.";
      if (inferred.klass.startsWith("Class I") || inferred.klass.startsWith("Class II")) return "Loss-of-function pattern; ETI-class benefit plausible if genotype aligns.";
      if (inferred.klass.startsWith("Class III")) return "Gating features; ivacaftor-class benefit plausible if genotype confirms.";
      if (inferred.klass.startsWith("Class IV") || inferred.klass.startsWith("Class V")) return "Milder classes; benefit may be genotype-specific.";
      return "Insufficient information to estimate eligibility.";
    }

    // --- Feature breakdown / Summary ---
    function featuresBreakdown(inputs){
      const {sweat, fev1, bmi, pseudo} = inputs;
      const cfrdYes = (val('cfrd')==="Yes");
      const piYes = (val('pi')==="Yes");
      const modSel = val('modulator');
      const sh = num('shannon');
      const dys = num('dysbiosis');
      const dom = val('dominant_taxa') || "—";
      const rich = val('path_richness') || "—";

      const sweatScore = (sweat===null?0:W.sweat*(sweat>=60?+18:(sweat>=30?+8:-6)));
      const fev1Score  = (fev1===null?0:W.fev1*(fev1<60?+12:(fev1>=90?-6:-2)));
      const pseudoScore= (pseudo==="Chronic"?W.pseudo*+8:(pseudo==="Intermittent"?W.pseudo*+4:0));
      const bmiScore   = (bmi===null?0:W.bmi*(bmi<18.5?+6:(bmi>=23?-4:0)));
      const comorbScore= (cfrdYes?W.cfrd*+4:0) + (piYes?W.pi*+3:0);
      const modScore   = (modSel? -6 : 0);

      let microScore = 0;
      if (sh!=null){ microScore += W.microdiv * ((2.0 - sh)*3.0); }
      if (dys!=null){ microScore += W.dysbio * ((dys/100)*6.0); }
      if (rich !== "—"){ microScore += W.rich * (parseInt(rich||"0") * 0.8); }
      if (dom==="Pseudomonas-dominant"){ microScore += 8; }
      else if (dom==="Achromobacter"){ microScore += 5; }
      else if (dom==="Staph-dominant"){ microScore += 3; }
      else if (dom==="Anaerobe-leaning"){ microScore += 2; }
      else if (dom==="Balanced/commensal"){ microScore -= 2; }

      const list = [
        {name:"Sweat chloride", val: (sweat===null?"—":sweat+" mmol/L"), score: Math.round(sweatScore)},
        {name:"FEV₁ % predicted", val: (fev1===null?"—":fev1+"%"), score: Math.round(fev1Score)},
        {name:"Pseudomonas status", val: val('pseudomonas')||"—", score: Math.round(pseudoScore)},
        {name:"Nutritional status (BMI)", val: (bmi===null?"—":bmi), score: Math.round(bmiScore)},
        {name:"Comorbidities (CFRD/PI)", val: ((cfrdYes?"CFRD ":"") + (piYes?"PI":"")) || "—", score: Math.round(comorbScore)},
        {name:"On CFTR modulator", val: val('modulator')||"No", score: Math.round(modScore)},
        {name:"Airway microbiome (composite)", val: `Shannon=${sh??"—"}, Dysbiosis=${dys??"—"}, Dominant=${dom}, Richness=${rich}`, score: Math.round(microScore)},
        {name:"Infection dynamics", val: `${val('infection_burden')||0}% year + ${val('exacerbations')||0} exacerbations`, score: infectionDynamicsPenalty()}
      ];
      return list;
    }

    function computeAUROC(pairs){
      const pts = pairs.slice().sort((a,b)=> b.pred - a.pred);
      let tp=0, fp=0;
      const P = pairs.filter(p=>p.label===1).length;
      const N = pairs.length - P;
      if (P===0 || N===0) return null;
      let prevPred = +Infinity;
      let auc = 0, fprPrev = 0, tprPrev = 0;
      pts.forEach(p=>{
        if (p.pred !== prevPred){
          auc += (fprPrev - (fp/N)) * (tprPrev + (tp/P)) / 2;
          tprPrev = tp/P; fprPrev = fp/N; prevPred = p.pred;
        }
        if (p.label===1) tp++; else fp++;
      });
      const tpr = tp/P, fpr = fp/N;
      auc += (1 - fprPrev) * (tprPrev + tpr) / 2;
      return clamp(auc, 0, 1);
    }
    function computeAUPRC(pairs){
      const pts = pairs.slice().sort((a,b)=> b.pred - a.pred);
      const P = pairs.filter(p=>p.label===1).length;
      if (P===0) return null;
      let tp=0, fp=0, prevPred=+Infinity;
      let area = 0, prevRec = 0;
      pts.forEach(p=>{
        if (p.pred !== prevPred){
          area += ((tp/(tp+fp) || 0)) * (((tp)/P) - prevRec);
          prevRec = (tp/P);
          prevPred = p.pred;
        }
        if (p.label===1) tp++; else fp++;
      });
      area += ((tp/(tp+fp) || 0)) * (1 - prevRec);
      return clamp(area, 0, 1);
    }
    function computeBrier(pairs){
      if (!pairs.length) return null;
      const mse = pairs.reduce((a,p)=> a + Math.pow(p.pred/100 - p.label, 2), 0) / pairs.length;
      return mse;
    }

    function effectPriorsFEV1(therapy, p){
      let mean = 0;
      const lowerClass = (p.genoClass||"").toLowerCase();
      const g = p.g.toLowerCase();
      if (therapy==="ETI"){
        if (g.includes("f508del") && (g.match(/f508del/g)||[]).length>=2) mean = 10.5;
        else if (g.includes("f508del")) mean = 7.5;
        else if (lowerClass.includes("class ii")) mean = 6.5;
        else if (lowerClass.includes("class i")) mean = 4.5;
        else if (lowerClass.includes("class iii")) mean = 5.0;
        else mean = 3.0;
      } else if (therapy==="Ivacaftor"){
        if (g.includes("g551d")) mean = 10.0;
        else if (g.includes("r117h")) mean = 4.0;
        else if (lowerClass.includes("class iii")) mean = 6.0;
        else mean = 2.0;
      } else if (therapy==="Tez/Iva"){
        if (g.includes("f508del")) mean = 3.5; else mean = 2.0;
      } else { mean = 0.0; }

      if (p.genoRef?.resp){
        const key = therapy==="ETI" ? "ETI" : (therapy==="Ivacaftor" ? "Iva" : "TezIva");
        const mult = p.genoRef.resp[key] ?? 1.0;
        mean *= (0.85 + 0.3*mult);
      }

      const sevAdj = clamp((80 - (Number.isFinite(p.baseFEV1)?p.baseFEV1:80)) / 40, -0.5, 0.5);
      mean += sevAdj * 4;

      let microMult = 1.0;
      if (p.sh!=null && p.sh < 2.0) microMult -= 0.08*(2.0 - p.sh);
      if ((p.dys||0) > 60) microMult -= 0.05;
      if ((p.dom||"")==="Pseudomonas-dominant") microMult -= 0.08;
      mean *= clamp(microMult, 0.7, 1.1);

      return mean;
    }

    function effectPriorsBMI(therapy, p){
      let mean = 0;
      const lower = (p.genoClass||"").toLowerCase(); const g = p.g.toLowerCase();
      if (therapy==="ETI"){
        mean = 1.5;
        if (g.includes("f508del") && (g.match(/f508del/g)||[]).length>=2) mean = 1.7;
        else if (g.includes("f508del")) mean = 1.4;
        else if (lower.includes("class i")) mean = 1.2;
      } else if (therapy==="Ivacaftor"){
        mean = 0.8; if (g.includes("g551d")) mean = 1.0;
      } else if (therapy==="Tez/Iva"){ mean = 0.5; } else { mean = 0.0; }

      if (p.genoRef?.resp){
        const key = therapy==="ETI" ? "ETI" : (therapy==="Ivacaftor" ? "Iva" : "TezIva");
        const mult = p.genoRef.resp[key] ?? 1.0;
        mean *= (0.9 + 0.25*mult);
      }

      if (p.agegrp==="child") mean += 0.4;
      if (p.agegrp==="adolescent") mean += 0.2;

      const bmiAdj = clamp((22 - (Number.isFinite(p.baseBMI)?p.baseBMI:22))/6, -0.3, 0.6);
      mean += bmiAdj * 0.8;

      if ((p.dys||0) > 70) mean -= 0.1;
      return mean;
    }

    function horizonScale(h){
      if (h<=6) return 0.7;

      if (evidence==="Limited") { sdFev = 2.8; sdBmi = 0.45; }
      if (isRare) { sdFev += 0.6; sdBmi += 0.1; }
      return { sdFev, sdBmi };
    }

    function compute(){
      if (!consentAck){ const banner = get("consentBanner"); if (banner) banner.style.display = "block"; alert("Acknowledge consent before computing."); return; }
      const g1 = val("variant1").toLowerCase(); const g2 = val("variant2").toLowerCase();
      const inputs = {
        sweat: num("sweat"), fev1: num("fev1"), age: num("age"), sex: val("sex"),
        bmi: num("bmi"), pseudo: val("pseudomonas"), ntm: val("ntm"),
        cfrd: val("cfrd")==="Yes", pi: val("pi")==="Yes", mod: val("modulator"),
        g1, g2, shannon: val("shannon"), dominant_taxa: val("dominant_taxa"),
        path_richness: val("path_richness"), dysbiosis: val("dysbiosis"),
        recent_abx: val("recent_abx"), polymicrobial: val("polymicrobial")
      };
      completenessBadge();
      const manualRare = val("rareFlag"); const evLevel = val("evidence");
      const rarity = detectRarity(g1,g2);
      const isRare = manualRare === "Yes" ? true : (manualRare === "No" ? false : rarity.autoRare);

      const genoRef = cftr2GenoReference(g1, g2);
      let inferred = inferGenotypeClass(inputs);
      inferred = refineWithGenotype(g1, g2, inferred);

      const scores = computeScores(inputs, genoRef);

      let confAdj = 0;
      if (isRare) confAdj -= 8;
      if (evLevel === "Limited") confAdj -= 5;
      else if (evLevel === "Moderate") confAdj -= 2;
      else if (evLevel === "Strong") confAdj += 4;
      if (genoRef) confAdj += 2;

      let eligibilityTxt = modulatorEligibility(g1,g2,inferred);
      if (isRare) eligibilityTxt += " Caution: rare/uncertain variant — eligibility depends on functional data and labeling.";

      get("riskScore").textContent = scores.risk;
      get("severityIndex").textContent = scores.sev;
      get("sevBar").style.width = scores.sev + "%";
      get("genoClass").textContent = inferred.klass;
      get("genoExamples").textContent = inferred.examples;
      const cVal = clamp(inferred.confidence + confAdj, 35, 98);
      get("confidence").textContent = cVal + "%";
      get("eligibility").textContent = eligibilityTxt;

      const rb = get("rarityBanner");
      if (rb) rb.style.display = isRare ? "block" : "none";

      const explain = featuresBreakdown(inputs);
      renderExplain(explain);

      fillSummary(inputs, scores, inferred);
      get("predictedProfile").textContent = profileText(scores.sev);
      get("mutationScore").textContent = inferred.mutationSeverity;
      get("mutBar").style.width = inferred.mutationSeverity + "%";

      const rationaleBits = [...inferred.signals];
      const sh = num('shannon'); const dys = num('dysbiosis'); const dom = val('dominant_taxa'); const rich = val('path_richness');
      if (dom) rationaleBits.push(`microbiome: ${dom.toLowerCase()}`);
      if (sh!=null) rationaleBits.push(`diversity ${sh<2 ? "low" : (sh>3.5?"high":"moderate")} (Shannon ${sh})`);
      if (dys!=null && dys>=60) rationaleBits.push(`dysbiosis high (${dys})`);
      if (rich) rationaleBits.push(`pathogen richness ${rich}`);
      if (genoRef?.sevAnchor!=null) rationaleBits.push(`CFTR2 prior anchor ~${genoRef.sevAnchor}/100`);

      get("rationale").textContent = rationaleBits.join(", ");

      // Run simulator immediately after compute
      updateSimAdherenceBadge();
      const simOut = simulateResponse();

      // Save to history
      saveHistory({
        ts: Date.now(),
        model: MODEL_VERSION,
        inputs,
        scores, geno: inferred, eligibility: eligibilityTxt
      });

      // Audit record
      auditRecord(inputs, scores, inferred, simOut);
      renderAudit();

      // Refresh report preview
      refreshReportPreview();
      updateQueueStatus();
    }

    async function sendQueued(){
      if (!session.signedIn) { alert("Sign in to send FHIR data."); return; }
      const baseUrl = val("fhirUrl");
      const token = val("fhirToken");
      if (!baseUrl || !token) { alert("Set FHIR URL and Bearer token first."); return; }
      let q = getQueue();
      if (!q.length) { alert("Queue is empty."); return; }
      const sent = [];
      for (const item of q){
        try {
          // DocumentReference (JSON report)

            resourceType: "DocumentReference",

            content: [{

                data: btoa(unescape(encodeURIComponent(JSON.stringify(item.report)))),
                title: `Stratacyst-Report-${item.ts}.json`
              }
            }]
          };
          await fetch(`${baseUrl.replace(/\/$/,"")}/DocumentReference`, {
            method: "POST",
            headers: { "Content-Type":"application/fhir+json","Authorization":`Bearer ${token}` },
            body: JSON.stringify(docRef)
          });

          // Observation (risk/severity summary)
          const obs = {
            resourceType: "Observation",
            status: "final",
            category: [{ text: "Risk assessment" }],

    async function runValidation(){
      const file = get("valCsv")?.files?.[0];
      if (!file) { alert("Select a CSV dataset first."); return; }
      const text = await file.text();
      const { rows } = parseCSV(text);
      if (!rows.length) { alert("No rows parsed from CSV."); return; }

      const pairs = [];
      rows.forEach(r=>{
        const out = (r.outcome||"").trim();
        const label = out==="" ? null : (parseInt(out) ? 1 : 0);
        const { pred } = computeRiskForRow(r);
        if (label!==null) pairs.push({ pred, label });
      });

      // Metrics (only if labels present)
      if (!pairs.length) {
        get("kpi_auroc").textContent = "—";
        get("kpi_auprc").textContent = "—";
        get("kpi_brier").textContent = "—";
        get("kpi_calslope").textContent = "—";
        get("val_reliability").innerHTML = "";
        get("val_decision").innerHTML = "";
        alert("No outcome labels found; computed scores only.");
        return;
      }

      const auroc = computeAUROC(pairs);
      const auprc = computeAUPRC(pairs);
      const brier = computeBrier(pairs);

      // Simple calibration slope via logistic regression approximation (binning)
      const bins = Array.from({length:10}, ()=>({ sumY:0, sumP:0, n:0 }));
      pairs.forEach(p=>{
        const b = Math.min(9, Math.max(0, Math.floor(p.pred/10)));
        bins[b].sumY += p.label; bins[b].sumP += (p.pred/100); bins[b].n += 1;
      });
      const pts = bins.filter(b=>b.n>0).map(b=>({ x: b.sumP/b.n, y: b.sumY/b.n }));
      let slope = "—";
      if (pts.length>=2){
        const xbar = pts.reduce((a,p)=>a+p.x,0)/pts.length;
        const ybar = pts.reduce((a,p)=>a+p.y,0)/pts.length;
        const num = pts.reduce((a,p)=> a + (p.x-xbar)*(p.y-ybar), 0);
        const den = pts.reduce((a,p)=> a + (p.x-xbar)*(p.x-xbar), 0) || 1;
        slope = round(num/den, 3);
      }

      get("kpi_auroc").textContent = (auroc==null ? "—" : round(auroc,3));
      get("kpi_auprc").textContent = (auprc==null ? "—" : round(auprc,3));
      get("kpi_brier").textContent = round(brier,3);
      get("kpi_calslope").textContent = (typeof slope==="number" ? slope : "—");

      drawReliability(pairs, "val_reliability");
      drawDecisionCurve(pairs, "val_decision");
      alert("Validation completed.");
    }
