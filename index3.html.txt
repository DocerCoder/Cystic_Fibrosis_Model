  <script>
    // Synonyms and normalization
    const VARIANT_SYNONYMS = {
      "F508DEL": "F508DEL", "DELF508": "F508DEL", "PHE508DEL": "F508DEL", "CFTR:F508DEL": "F508DEL",
      "G551D": "G551D", "W1282X": "W1282X", "G542X": "G542X", "R553X": "R553X", "N1303K": "N1303K",
      "R117H": "R117H", "A455E": "A455E", "D1152H": "D1152H"
    };

    function normalizeHGVS(raw) {
      let s = (raw || "").trim().replace(/>/g, "→").replace(/\s+/g, "");
      s = s.replace(/^CFTR:/i, "").replace(/^c\./i, "").replace(/^p\./i, "");
      s = s.replace(/^PHE?508DEL$/i, "F508del");
      return s;
    }

    function canonicalizeVariant(raw) {
      if (!raw) return "";
      const norm = normalizeHGVS(raw);
      const candidates = [norm, norm.toUpperCase(), norm.toLowerCase()];
      const legacy = Catalog.legacyMap || {};
      for (const cand of candidates) { if (legacy[cand]) return legacy[cand]; }
      for (const cand of candidates) { const key = cand.toUpperCase(); if (VARIANT_SYNONYMS[key]) return VARIANT_SYNONYMS[key]; }
      const keys = Catalog.cftr2Meta ? Object.keys(Catalog.cftr2Meta) : [];
      for (const cand of candidates) { if (Catalog.cftr2Meta?.[cand]) return cand; }
      for (const key of keys) { if (key.toLowerCase() === norm.toLowerCase()) return key; }
      return norm;
    }

    function getCoreAllele(variant) {
      if (!variant) return "";
      const root = String(variant).split(/[^A-Za-z0-9→]+/)[0];
      return root.toUpperCase();
    }

    function assessVariants(raw1, raw2) {
      const lists = Catalog.genotype?.genotypes?.modulator_eligibility_lists || {};
      const toRootSet = (arr = []) => new Set(arr.map(v => getCoreAllele(canonicalizeVariant(v))));
      const ETI = toRootSet(lists.FDA_ETI_responsive);
      const IVA = toRootSet(lists.Ivacaftor_responsive);
      const FR  = toRootSet(lists.French_compassionate_use);
      return [raw1, raw2].filter(Boolean).map(raw => {
        const canonical = canonicalizeVariant(raw);
        const root = getCoreAllele(canonical);
        const meta = Catalog.cftr2Meta?.[canonical] || {};
        return {
          raw, canonical, root,
          isCfCausing: Boolean(meta.is_cf_causing),
          lists: { ETI: ETI.has(root), IVA: IVA.has(root), French: FR.has(root) }
        };
      });
    }

    function classifyModulatorGroup(alleles) {
      const roots = alleles.map(a => a.root);
      const f508Count = roots.filter(r => r === "F508DEL").length;
      const hasETI = alleles.some(a => a.lists.ETI);
      const hasIVA = alleles.some(a => a.lists.IVA);
      if (f508Count >= 2) return { group: "group_1", reason: "F508del homozygous" };
      if (f508Count === 1) return { group: "group_2", reason: "F508del heterozygous" };
      if (hasETI)          return { group: "group_3", reason: "ETI-responsive allele" };
      if (hasIVA)          return { group: "group_4", reason: "Ivacaftor-responsive allele" };
      return { group: "group_5", reason: "No known responsive allele" };
    }

    function computeCftrFunctionScore(rootAllele) {
      const f = Catalog.cftrFunction?.[rootAllele];
      if (!f) return null;
      const score = (f.gating || 0) * (f.cond || 0) * (f.proc || 0);
      return Math.min(1, Math.max(0, score));
    }

    const CFTR2_PRIORS = {
      F508DEL: { class: "II",  sevPrior: 0.82, resp: { ETI:1.0, TezIva:0.4, Iva:0.1 } },
      G551D:   { class: "III", sevPrior: 0.60, resp: { ETI:0.7, TezIva:0.2, Iva:1.0 } },
      W1282X:  { class: "I",   sevPrior: 0.85, resp: { ETI:0.5, TezIva:0.2, Iva:0.1 } },
      G542X:   { class: "I",   sevPrior: 0.83, resp: { ETI:0.5, TezIva:0.2, Iva:0.1 } },
      R553X:   { class: "I",   sevPrior: 0.82, resp: { ETI:0.5, TezIva:0.2, Iva:0.1 } },
      N1303K:  { class: "II",  sevPrior: 0.78, resp: { ETI:0.9, TezIva:0.3, Iva:0.1 } },
      R117H:   { class: "IV",  sevPrior: 0.40, resp: { ETI:0.6, TezIva:0.3, Iva:0.7 } },
      A455E:   { class: "IV",  sevPrior: 0.45, resp: { ETI:0.6, TezIva:0.3, Iva:0.4 } },
      D1152H:  { class: "IV",  sevPrior: 0.42, resp: { ETI:0.6, TezIva:0.3, Iva:0.5 } }
    };

    function cftr2Reference(raw1, raw2) {
      const roots  = [raw1, raw2].filter(Boolean).map(r => getCoreAllele(canonicalizeVariant(r)));
      const priors = roots.map(r => CFTR2_PRIORS[r]).filter(Boolean);
      if (!priors.length) return null;
      const worst = priors.reduce((a,b) => b.sevPrior > a.sevPrior ? b : a);
      return { sevAnchor: Math.round(worst.sevPrior * 100), classHint: worst.class, resp: worst.resp };
    }

    function detectVariantRarity(raw1, raw2) {
      const roots   = [raw1, raw2].filter(Boolean).map(r => getCoreAllele(canonicalizeVariant(r)));
      const known   = new Set(Object.keys(CFTR2_PRIORS));
      const anyKnown= roots.some(r => known.has(r));
      return { autoRare: !anyKnown, reason: anyKnown ? "known catalog match" : "not in known catalog" };
    }

    function computeMicrobiomeImpact(inputs) {
      let riskΔ = 0, sevΔ = 0;
      if (inputs.shannon != null) {
        const d = 2.0 - inputs.shannon;
        riskΔ += d * 2.5; sevΔ += d * 3.0;
      }
      if (inputs.dysbiosis != null) {
        riskΔ += (inputs.dysbiosis / 100) * 4.0;
        sevΔ  += (inputs.dysbiosis / 100) * 6.0;
      }
      if (inputs.pathRichness != null) {
        riskΔ += inputs.pathRichness * 0.6;
        sevΔ  += inputs.pathRichness * 0.8;
      }
      switch (inputs.dominantTaxa) {
        case 'Pseudomonas-dominant': riskΔ += 4;  sevΔ += 8;  break;
        case 'Achromobacter':        riskΔ += 3;  sevΔ += 5;  break;
        case 'Staph-dominant':       riskΔ += 1.5;sevΔ += 2.5;break;
        case 'Anaerobe-leaning':     riskΔ += 1;  sevΔ += 2;  break;
        case 'Balanced/commensal':   riskΔ -= 1;  sevΔ -= 2;  break;
      }
      if (inputs.recentAbx === 'Yes (≤30 days)') { riskΔ -= 1; sevΔ -= 2; }
      if (inputs.polymicrobial === 'Yes') { riskΔ += 1.5; sevΔ += 2.5; }
      return { riskΔ: Math.round(riskΔ), sevΔ: Math.round(sevΔ) };
    }

    function computeInfectionPenalty(inputs) {
      const burden = inputs.infection_burden || 0;
      const exac   = inputs.exacerbations   || 0;
      const erad   = inputs.eradication === 'Yes' ? -2 : 0;
      return Math.round((burden / 100) * 8 + Math.min(exac,6) * 1.5 + erad);
    }

    function computeHeuristicScores(inputs) {
      const w = {
        sweat:  parseFloat(document.getElementById('w_sweat').value)  || 1,
        fev1:   parseFloat(document.getElementById('w_fev1').value)   || 1,
        bmi:    parseFloat(document.getElementById('w_bmi').value)    || 1,
        pseudo: parseFloat(document.getElementById('w_pseudo').value) || 1
      };

      const alleles = assessVariants(inputs.variant1, inputs.variant2);
      const cfCount = alleles.filter(a => a.isCfCausing).length;

      let risk = 20;
      if (inputs.sweat != null) {
        risk += w.sweat * (inputs.sweat >= 60 ? 25 : inputs.sweat >= 30 ? 10 : -10);
      }
      if (inputs.fev1 != null) {
        risk += w.fev1 * (inputs.fev1 < 60 ? 10 : inputs.fev1 >= 90 ? -5 : 0);
      }
      if (inputs.pseudomonas === 'Chronic') risk += w.pseudo * 6;
      else if (inputs.pseudomonas === 'Intermittent') risk += w.pseudo * 3;

      if (cfCount >= 2) risk += 10; else if (cfCount === 1) risk += 6;

      if (inputs.bmi != null) {
        risk += w.bmi * (inputs.bmi < 18.5 ? 8 : inputs.bmi >= 23 ? -5 : 0);
      }
      if (inputs.cfrd) risk += 6;
      if (inputs.pi)   risk += 4;

      const { riskΔ } = computeMicrobiomeImpact(inputs);
      risk += riskΔ;

      if (inputs.priorAnchor != null) {
        risk = Math.round(risk + (inputs.priorAnchor - 50) / 40);
      }

      let sev = 50;
      if (inputs.fev1 != null) sev += (80 - inputs.fev1) * 0.5;
      if (inputs.pseudomonas === 'Chronic') sev += 10;
      if (inputs.bmi < 18.5) sev += 8;
      if (inputs.modulator) sev -= 6;
      sev += computeInfectionPenalty(inputs);

      return {
        risk: Math.min(100, Math.max(0, Math.round(risk))),
        sev:  Math.min(100, Math.max(0, Math.round(sev)))
      };
    }

    function selectAgeBand(age, bands) {
      const a = parseInt(age,10);
      for (let b of bands) {
        const m = b.match(/^(\d+)(?:-(\d+)|\+)?$/);
        if (!m) continue;
        const lo = parseInt(m[1],10);
        const hi = m[2] ? parseInt(m[2],10) : Infinity;
        if (a >= lo && a <= hi) return b;
      }
      return bands[0];
    }

    function computeQuantilePercentile(age, sex, fev1) {
      const tables = Catalog.lung?.lung_function;
      if (!tables || fev1 == null) return null;

      const annualGroups = tables.fev1_annual_review?.by_age_group || {};
      const bands = Object.keys(annualGroups);
      if (!bands.length) return null;

      const band = selectAgeBand(age ?? 26, bands);
      const annual = tables.fev1_annual_review.by_age_group[band] || {};
      const best   = tables.fev1_best?.by_age_group?.[band] || {};

      const sexKey = String(sex || "overall").toLowerCase();
      const mu  = annual[sexKey] ?? annual.overall ?? 75;
      const hi  = best[sexKey]   ?? best.overall   ?? Math.max(80, mu + 8);
      const lo  = Math.max(0, mu - (hi - mu || 10));

      if (fev1 <= lo) return 0;
      let p;
      if (fev1 <= mu) p = Math.pow(fev1 - lo,2)/((mu - lo)*(hi - lo));
      else if (fev1 < hi) p = 1 - Math.pow(hi - fev1,2)/((hi - lo)*(hi - mu));
      else p = 1;

      return Math.round(Math.max(0, Math.min(1, p)) * 100);
    }

    function computeSKScore(inputs) {
      let A = 25 - Math.min(inputs.exacerbations || 0,6)
              - (inputs.pseudomonas==='Chronic'?6:inputs.pseudomonas==='Intermittent'?3:0);
      A = Math.max(5, Math.min(25, A));
      let P = 25;
      if (inputs.fev1 != null) {
        if (inputs.fev1 < 40)      P = 8;
        else if (inputs.fev1 < 60) P = 14;
        else if (inputs.fev1 < 80) P = 18;
        else                       P = 25;
      }
      let N = 25;
      if (inputs.bmi != null) {
        if (inputs.bmi < 18.5)      N -= 8;
        else if (inputs.bmi < 21)   N -= 4;
        else if (inputs.bmi >= 25)  N -= 2;
      }
      if (inputs.cfrd) N -= 3;
      if (inputs.pi)   N -= 3;
      N = Math.max(4, Math.min(25, N));
      let R = 25;
      const sevIndex = inputs.sevIndex;
      if (!isNaN(sevIndex)) {
        if (sevIndex >= 70)      R = 8;
        else if (sevIndex >= 55) R = 14;
        else if (sevIndex >= 40) R = 18;
        else                     R = 22;
      } else if (inputs.fev1 != null) {
        R = inputs.fev1 < 40 ? 8 : inputs.fev1 < 60 ? 14 : inputs.fev1 < 80 ? 18 : 22;
      }
      return Math.round(Math.min(100,Math.max(0,A+P+N+R)));
    }

    function bayesianPosteriorRisk(inputs, genoRef) {
      let logit = 0;

      if (inputs.sweat != null) {
        logit += inputs.sweat >= 90 ? 3.0
             :  inputs.sweat >= 60 ? 1.8
             :  inputs.sweat >= 30 ? 0.7
             : -1.2;
      }

      const alleles = assessVariants(inputs.variant1, inputs.variant2);
      const cfCount = alleles.filter(a => a.isCfCausing).length;
      if (cfCount >= 2) logit += 2.2;
      else if (cfCount === 1) logit += 1.2;

      if (genoRef?.sevAnchor != null) {
        logit += (genoRef.sevAnchor - 50) / 18;
      }

      if (inputs.fev1 != null) {
        logit += inputs.fev1 < 40 ? 1.0
             :  inputs.fev1 < 60 ? 0.6
             :  inputs.fev1 >= 90 ? -0.4
             :  0;
      }

      logit += inputs.pseudomonas === 'Chronic'      ? 0.7
            :  inputs.pseudomonas === 'Intermittent' ? 0.3
            :  inputs.pseudomonas === 'None'         ? -0.2
            :  0;

      if (inputs.bmi != null && inputs.bmi < 18.5) logit += 0.5;
      if (inputs.cfrd) logit += 0.4;
      if (inputs.pi)   logit += 0.6;

      if (inputs.shannon != null && inputs.shannon < 2.0) {
        logit += 0.3 * (2.0 - inputs.shannon);
      }
      if ((inputs.dysbiosis || 0) >= 70) logit += 0.4;
      if (inputs.dominantTaxa === 'Pseudomonas-dominant') logit += 0.5;

      const fRoots = alleles.map(a => a.root);
      const fScores = fRoots.map(r => computeCftrFunctionScore(r) || 0);
      const cftrPrior = fScores.length ? Math.max(...fScores) : 0;
      if (cftrPrior > 0) logit += (cftrPrior - 0.5) * 1.2;

      const rarity = detectVariantRarity(inputs.variant1, inputs.variant2);
      if (rarity.autoRare) logit -= 0.6;
      if (inputs.evidence === 'Strong') logit += 0.5;
      else if (inputs.evidence === 'Limited') logit -= 0.4;

      logit = Math.max(-10, Math.min(10, logit));
      const p = 1 / (1 + Math.exp(-logit));
      const bounded = Math.max(0.001, Math.min(0.999, p));
      return Math.round(bounded * 1000) / 1000;
    }

    function uniqueSorted(arr){ return Array.from(new Set(arr)).sort((a,b)=>b-a); }
    function computeAUROC(labels,scores) {
      let auc=0, prevFPR=0, prevTPR=0;
      uniqueSorted(scores).forEach(t=>{
        let tp=0,fp=0,tn=0,fn=0;
        labels.forEach((y,i)=> {
          const p=scores[i]>=t;
          if(p&&y) tp++; else if(p&&!y) fp++;
          else if(!p&&y) fn++; else tn++;
        });
        const tpr=tp/(tp+fn||1), fpr=fp/(fp+tn||1);
        auc += (fpr-prevFPR)*(tpr+prevTPR)/2;
        prevFPR=fpr; prevTPR=tpr;
      });
      return Math.max(0,1-auc);
    }
    function computeAUPRC(labels,scores) {
      let prevR=0, prevP=1, auprc=0;
      uniqueSorted(scores).forEach(t=>{
        let tp=0,fp=0,fn=0;
        labels.forEach((y,i)=> {
          const p=scores[i]>=t;
          if(p&&y) tp++; if(p&&!y) fp++; if(!p&&y) fn++;
        });
        const prec=tp/(tp+fp||1), rec=tp/(tp+fn||1);
        auprc += (rec-prevR)*(prec+prevP)/2;
        prevR=rec; prevP=prec;
      });
      return auprc;
    }
    function computeBrier(labels,scores) {
      let sum=0;
      labels.forEach((y,i)=> sum += Math.pow(scores[i]-y,2));
      return sum/labels.length;
    }
    function computeCalibration(labels,scores) {
      const safe = scores.map(p => Math.max(0.001, Math.min(0.999, p)));
      const xs = safe.map(p => Math.log(p / (1 - p)));
      const ys = labels.map(y => y ? 1 : 0);
      const n = xs.length;
      if (!n) return { slope: 0, intercept: 0 };
      const mx = xs.reduce((a,b)=>a+b,0)/n, my = ys.reduce((a,b)=>a+b,0)/n;
      let num=0, den=0;
      for(let i=0;i<n;i++){
        num += (xs[i]-mx)*(ys[i]-my);
        den += (xs[i]-mx)*(xs[i]-mx);
      }
      const slope = den ? num/den : 0;
      const intercept = my - slope*mx;
      return { slope, intercept };
    }

      const results = {
        model: MODEL_VERSION,
        risk: heur.risk,
        severity: heur.sev,
        bayesianRisk: bayesPct,
        severityPercentile: sevPercentile,
        skScore: sk,
        confidence: Math.min(95, Math.max(5, Math.round(50 + (bayes - 0.5) * 80))),
        genotypeClass: genoClass,
        eligibility,
        alleles: allelesAssessed.map(a => ({
          root: a.root,
          cfCausing: a.isCfCausing,
          ETI: a.lists.ETI,
          IVA: a.lists.IVA
        }))
      };

      lastInputs = inputs;
      lastResults = results;

      // --- UI updates (guarded) ---
      const safeSet = (id, val) => { const el = document.getElementById(id); if (el) el.textContent = val; };
      safeSet('bayesRisk', results.bayesianRisk);
      safeSet('severityIndex', results.severity);
      safeSet('sevPercentile', results.severityPercentile ?? 'Not available');
      safeSet('skScore', results.skScore);
      safeSet('genoClass', `Genotype class: ${results.genotypeClass}`);
      safeSet('eligibilityText', results.eligibility);
      safeSet('primarySummary',
        `Bayesian risk ${results.bayesianRisk}/100; Severity ${results.severity}/100; Percentile ${results.severityPercentile ?? '—'}; SK ${results.skScore}.`);

      const rp = document.getElementById('riskPointer'); if (rp) rp.style.left = `${results.bayesianRisk}%`;
      const sp = document.getElementById('sevPointer');  if (sp) sp.style.left = `${results.severity}%`;

      // --- Details rationale ---
      const rationale = [];
      if (inputs.sweat != null) rationale.push(`Sweat chloride ${inputs.sweat} mmol/L anchors CF risk`);
      if (inputs.fev1 != null) rationale.push(`FEV₁ ${inputs.fev1}% influences severity and SK`);
      if (inputs.bmi != null) rationale.push(`BMI ${inputs.bmi} adjusts nutrition component`);
      if (inputs.pseudomonas) rationale.push(`Pseudomonas: ${inputs.pseudomonas}`);
      if (inputs.cfrd) rationale.push(`CFRD present`);
      if (inputs.pi) rationale.push(`Pancreatic insufficiency`);
      if (inputs.variant1 || inputs.variant2) rationale.push(`Genotype: ${results.genotypeClass}`);
      if (inputs.shannon != null) rationale.push(`Shannon diversity ${inputs.shannon}`);
      if (inputs.dysbiosis != null) rationale.push(`Dysbiosis ${inputs.dysbiosis}%`);
      const rationalePre = document.getElementById('detailRationale'); if (rationalePre) rationalePre.textContent = rationale.join('\n');
      const inputsPre = document.getElementById('detailInputs'); if (inputsPre) inputsPre.textContent = JSON.stringify(inputs, null, 2);

      // --- History & audit ---
      saveHistory(inputs, {
        bayesianRisk: results.bayesianRisk,
        severity: results.severity,
        genotypeClass: results.genotypeClass,
        eligibility: results.eligibility,
        model: results.model
      });
      auditRecord(inputs, results, session.role);
      refreshReportPreview();

      // --- Show results card ---
      const rc = document.getElementById('resultsCard'); if (rc) rc.style.display = 'block';
    }

    function exportReport(type) {
      const data = buildReportData();
      if (!('risk' in data.results)) { alert('Compute a patient first.'); return; }

      if (type === 'json') {
        const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `stratacyst-report-${new Date().toISOString()}.json`;
        a.click();
        URL.revokeObjectURL(url);
        return;
      }

      if (type === 'pdf') {
        const redactNames = document.getElementById('redactNames')?.checked ?? true;
        const redactDates = document.getElementById('redactDates')?.checked ?? true;

        const inputs = JSON.parse(JSON.stringify(data.inputs));
        const meta   = JSON.parse(JSON.stringify(data.meta));

        if (redactDates) {
          meta.generatedAt = "REDACTED";
          if (inputs.notes) inputs.notes = String(inputs.notes)
            .replace(/\b\d{4}-\d{2}-\d{2}\b/g, "[REDACTED_DATE]")
            .replace(/\b(?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Sept|Oct|Nov|Dec)\b \d{1,2}, \d{4}/gi, "[REDACTED_DATE]");
        }
        if (redactNames) {
          for (const k of Object.keys(inputs)) {
            if (/name|mrn|identifier|uid|pid/i.test(k) && typeof inputs[k] === 'string') {
              inputs[k] = "[REDACTED]";
            }
          }
        }

        const w = window.open('', '_blank');
        const css = `
          body { font-family: Inter, system-ui, -apple-system, Segoe UI; color: #0b1020; margin: 24px; }
          h2 { margin: 0 0 6px 0; }
          .meta { font-size: 12px; color: #444; margin: 2px 0; }
          .kpis { display: grid; grid-template-columns: repeat(3, 1fr); gap: 8px; margin-top: 10px; }
          .kpi { border: 1px solid #ccd; border-radius: 8px; padding: 8px; }
          .kpi b { display: block; margin-bottom: 4px; }
          pre { background: #f6f8fb; border: 1px solid #dde; border-radius: 8px; padding: 10px; font-size: 12px; overflow: auto; }
          hr { margin: 12px 0; }
        `;
        const html = `
          <html><head><title>Stratacyst Report</title><style>${css}</style></head><body>
          <h2>${meta.title}</h2>
          <div class="meta">Model: ${meta.model} (${meta.schema})</div>
          <div class="meta">Generated: ${meta.generatedAt}</div>
          <div class="meta">InputHash: ${meta.inputHash} &nbsp; OutputHash: ${meta.outputHash}</div>
          <hr/>
          <div class="kpis">
            <div class="kpi"><b>Bayesian risk</b>${data.results.bayesianRisk}</div>
            <div class="kpi"><b>Severity index</b>${data.results.severity}</div>
            <div class="kpi"><b>Severity percentile</b>${data.results.severityPercentile ?? '—'}</div>
            <div class="kpi"><b>SK score</b>${data.results.skScore}</div>
            <div class="kpi"><b>Genotype class</b>${data.results.genotypeClass}</div>
            <div class="kpi"><b>Eligibility</b>${data.results.eligibility}</div>
          </div>
          <hr/>
          <b>Inputs</b>
          <pre>${JSON.stringify(inputs, null, 2)}</pre>
          </body></html>
        `;
        w.document.write(html);
        w.document.close();
        w.focus();
        w.print();
      }
    }

    function queueFHIR() {
      if (!lastInputs || !lastResults) { alert('Compute a patient first.'); return; }

    async function testFHIR() {
      const url = document.getElementById('fhirUrl')?.value?.trim();
      const token = document.getElementById('fhirToken')?.value?.trim();
      if (!url || !token) { alert('Enter FHIR URL and bearer token.'); return; }
      try {
        const resp = await fetch(url.replace(/\/+$/, '') + '/metadata', {
          method: 'GET',
          headers: { 'Authorization': `Bearer ${token}` }
        });
        alert(resp.ok ? 'FHIR connection OK.' : `FHIR test failed: ${resp.status}`);
      } catch (e) {
        alert('FHIR test failed: ' + (e?.message || 'unknown error'));
      }
    }
    async function sendQueued() {
      if (!FHIR_QUEUE.length) { alert('Queue is empty.'); return; }
      const sent = [];
      for (const item of FHIR_QUEUE) {
        const { cfg, reportData } = item;
        const base = (cfg.url || '').replace(/\/+$/, '');
        const token = cfg.token || '';
        const patientRef = cfg.patient || 'Patient/unknown';

        const obs = {
          resourceType: 'Observation',
          status: 'final',
          code: { coding: [{ system: 'http://loinc.org', code: 'LP123456-7', display: 'CF risk/severity (Stratacyst)' }] },
          subject: { reference: patientRef },
          effectiveDateTime: new Date().toISOString(),
          valueString: `Risk:${reportData.results.bayesianRisk}; Severity:${reportData.results.severity}; SK:${reportData.results.skScore}; Percentile:${reportData.results.severityPercentile ?? 'NA'}`,
          note: [{ text: `Model ${reportData.meta.model} (${reportData.meta.schema}) IH:${reportData.meta.inputHash} OH:${reportData.meta.outputHash}` }]
        };
        const doc = {
          resourceType: 'DocumentReference',
          status: 'current',
          type: { coding: [{ system: 'http://loinc.org', code: '11502-2', display: 'Clinical report' }] },
          subject: { reference: patientRef },
          date: new Date().toISOString(),
          content: [{
            attachment: {
              contentType: 'application/json',
              data: btoa(unescape(encodeURIComponent(JSON.stringify(reportData)))),
              title: `Stratacyst Report ${new Date().toISOString()}`
            }
          }]
        };
        try {
          const r1 = await fetch(base + '/Observation', {
            method: 'POST',
            headers: { 'Content-Type': 'application/fhir+json', 'Authorization': `Bearer ${token}` },
            body: JSON.stringify(obs)
          });
          const r2 = await fetch(base + '/DocumentReference', {
            method: 'POST',
            headers: { 'Content-Type': 'application/fhir+json', 'Authorization': `Bearer ${token}` },
            body: JSON.stringify(doc)
          });
          if (r1.ok && r2.ok) sent.push(item);
        } catch (e) { /* continue batch */ }
      }
      FHIR_QUEUE = FHIR_QUEUE.filter(x => !sent.includes(x));
      const el = document.getElementById('queueStatus');
      if (el) el.textContent = `Queue: ${FHIR_QUEUE.length}`;
      alert(`Sent ${sent.length} item(s).`);
    }

    // Wizard navigation (hardened)
    (function(){
      const TOTAL_STEPS = 4;
      const STEP_NAMES = { 1:"Phenotype",2:"Genotype",3:"Infection dynamics",4:"Modulator & Notes",5:"Results" };
      let currentStep = 1;

      function el(id){ return document.getElementById(id); }
      function qs(sel){ return document.querySelector(sel); }
      function qsa(sel){ return Array.from(document.querySelectorAll(sel)); }

      function showStep(step) {
        currentStep = step;
        const wizardCard  = qs(".wizard-card");
        const resultsCard = el("resultsCard");
        const steps       = qsa(".wizard-body .step");

        if (!wizardCard || !resultsCard || steps.length === 0) return;

        if (step <= TOTAL_STEPS) {
          wizardCard.style.display  = "block";
          resultsCard.style.display = "none";
          steps.forEach(el => { el.style.display = (parseInt(el.dataset.step,10) === step) ? "block" : "none"; });
        } else {
          wizardCard.style.display  = "none";
          resultsCard.style.display = "block";
        }
        updateTracker(); updateButtons(); updateCompleteness();
      }

      function updateTracker() {
        const tracker = el("stepTracker");
        if (!tracker) return;
        tracker.textContent = (currentStep <= TOTAL_STEPS)
          ? `Step ${currentStep} of ${TOTAL_STEPS}: ${STEP_NAMES[currentStep]}`
          : STEP_NAMES[5];
      }

      function updateButtons() {
        const backBtn = el("wizardBack");
        const nextBtn = el("wizardNext");
        if (!backBtn || !nextBtn) return;
        backBtn.disabled = (currentStep === 1);
        if (currentStep < TOTAL_STEPS) {
          nextBtn.textContent = "Next →"; nextBtn.setAttribute("aria-label","Next step"); nextBtn.disabled = false;
        } else if (currentStep === TOTAL_STEPS) {
          nextBtn.textContent = "Compute"; nextBtn.setAttribute("aria-label","Compute results"); nextBtn.disabled = false;
        } else {
          backBtn.style.display = "none"; nextBtn.style.display = "none";
        }
      }

      function updateCompleteness() {
        const completeness = el("completeness");
        if (!completeness) return;
        const required = [
          "sweat","fev1","bmi","pseudomonas","cfrd","pi",
          "infection_burden","exacerbations","eradication",
          "shannon","dysbiosis","dominantTaxa",
          "modulator"
        ];
        const know = el('knowGenotype')?.checked;
        if (know) required.push("variant1");

        let filled = 0;
        required.forEach(id => { const f = el(id); if (f && f.value) filled++; });
        const pct = Math.round(filled / required.length * 100);
        completeness.textContent = `Completeness: ${pct}%`;
        completeness.classList.toggle("ok",   pct >= 75);
        completeness.classList.toggle("warn", pct >= 40 && pct < 75);
        completeness.classList.toggle("bad",  pct < 40);
      }

      function onNext() {
        if (currentStep < TOTAL_STEPS) { showStep(currentStep + 1); }
        else if (currentStep === TOTAL_STEPS) { try { compute(); } catch(e){ console.error(e); alert('Compute failed: ' + e.message); } showStep(TOTAL_STEPS + 1); }
      }
      function onBack() {
        if (currentStep > 1 && currentStep <= TOTAL_STEPS + 1) { showStep(currentStep - 1); }
      }

      document.addEventListener("DOMContentLoaded", () => {
        const backBtn = el("wizardBack");
        const nextBtn = el("wizardNext");
        const computeBtn = el("computeBtn");

        if (backBtn) backBtn.addEventListener("click", onBack);
        if (nextBtn) nextBtn.addEventListener("click", onNext);
        if (backBtn) backBtn.addEventListener("keydown", e => { if (e.key==="Enter") onBack(); });
        if (nextBtn) nextBtn.addEventListener("keydown", e => { if (e.key==="Enter") onNext(); });
        if (computeBtn) computeBtn.addEventListener("click", () => { try { compute(); } catch(e){ console.error(e); alert('Compute failed: ' + e.message); } showStep(TOTAL_STEPS + 1); });

        showStep(1);
      });
    })();
